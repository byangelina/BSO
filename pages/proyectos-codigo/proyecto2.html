Julia

                    
                        # Comentarios en Julia comienzan con el símbolo '#' y pueden ser de una sola línea o multilinea.
                        
                        # Definición de una función en Julia
                        function saludar(nombre)
                        println("¡Hola, $nombre!")
                        end
                        
                        # Llamada a la función 'saludar'
                        saludar("Juan")
                        
                        # Definición de un bucle 'for' en Julia
                        for i in 1:5
                        println("El valor de i es: $i")
                        end
                        
                        # Definición de una estructura de control 'if-else' en Julia
                        edad = 25
                        if edad >= 18
                        println("Eres mayor de edad.")
                        else
                        println("Eres menor de edad.")
                        end
                        
                        # Definición de un arreglo en Julia
                        arreglo = [1, 2, 3, 4, 5]
                        
                        # Iteración sobre los elementos de un arreglo
                        for elemento in arreglo
                        println("El elemento actual es: $elemento")
                        end
